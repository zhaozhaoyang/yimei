1.0 用户登陆
请求格式
        {
            cmd:”userLogin”
            phone:""//用户手机号
			password：//密码
        }
返回格式
        {
            result:"0" //0未注册 1已注册
            resultNote:”失败原因”
			uid://
        }

请求格式
        {
            cmd:forgetPassWord
			phone：//用户手机号
			password：//密码
        }
 返回格式
        {
            result:"0" //0成功 1失败
            resultNote:”失败原因”
        }
