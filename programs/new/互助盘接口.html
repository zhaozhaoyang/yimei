1 登陆、注册

1.0 用户登陆
请求格式
        {
            cmd:”userLogin”
            phone:""//用户手机号
			password：//密码
        }
返回格式
        {
            result:"0" //0未注册 1已注册
            resultNote:”失败原因”
			uid://
        }
1.1 忘记密码
请求格式
        {
            cmd:forgetPassWord
			phone：//用户手机号
			password：//密码
        }
 返回格式
        {
            result:"0" //0成功 1失败
            resultNote:”失败原因”
        }
1.3 注册
请求格式
        {
            cmd:register
			nickName://昵称
			email://邮箱
			phone：//用户手机号
			password：//密码
			parentPhone://推荐人手机号
			bankId：//银行id
			bankNumber：//银行卡号
			branch;		// 支行名字      
			userName;		// 用户名字  
			alipay;		// 支付宝账号     
			remark;		// 备注        
        }
 返回格式
        {
            result:"0" //0成功 1失败
            resultNote:”失败原因”
        }
1.2 获取证码
请求格式

            cmd:”sendSms”

        }
返回格式
        {
			code：//

5.0 获取个人信息
请求格式
    {
        cmd:"getUserInfo"
		 uid:""//
    }
返回格式
    {
        result:"0" //0成功1失败
        resultNote:"失败原因"
		phone：//用户手机号
		icon：//头像
		nickName://昵称
		drawsNumber：//抽奖次数
		email://邮箱
		flag：//是否激活(1:是2：否)
		time://注册时间
		isOpen：//是否开启自动预约(1：是，0：否)
		isHost://是否托管(1:是0:否)
		parentId://领导id
		parentName://领导名字
		parentEmail://领导账号
		parentPhone://领导手机号
		integrityCoin;		// 诚信币 
		integrityValue;		// 诚信值 
		score;		// 积分       
		principal;		// 本息
		reward;		// 奖励    
		number;		// 激活券数量
		bankNumber;		// 银行卡号   
		branch;		// 支行名字       
		userName;		// 用户名字   
		alipay;		// 支付宝账号      
		remark;		// 备注 
		num;		// 排单剩余轮数      
        money;		// 排单额度        
        
		
}
